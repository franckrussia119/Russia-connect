# Highly resilient single-stage build for Coolify/PaaS
FROM node:20

# Force development environment during build to ensure devDependencies are installed
ENV NODE_ENV=development

WORKDIR /app

# Copy dependency files
COPY package*.json ./

# Install all dependencies (including devDependencies)
RUN npm install

# Copy source code files
COPY . .

# Build the application (runs vite build and bundles server.ts)
RUN npm run build

# Set production environment variables for runtime
ENV NODE_ENV=production
ENV PORT=3000

# Expose the port
EXPOSE 3000

# Run the bundled server file directly
CMD ["node", "dist/server.cjs"]
