# Stage 1: Build Stage
FROM node:20-slim AS builder

WORKDIR /app

# Ensure devDependencies are installed even if build environment has NODE_ENV=production
ENV NODE_ENV=development

# Copy dependency files
COPY package*.json ./

# Install all dependencies (including devDependencies)
RUN npm ci --include=dev

# Copy source code files
COPY . .

# Build the application (runs vite build and bundle server.ts)
RUN npm run build

# Stage 2: Runtime Stage
FROM node:20-slim AS runner

WORKDIR /app

# Install production-only dependencies
COPY package*.json ./
RUN npm ci --only=production

# Copy built assets and server bundles from builder stage
COPY --from=builder /app/dist ./dist

# Set production environment variables
ENV NODE_ENV=production
ENV PORT=3000

# Expose the configured port
EXPOSE 3000

# Run the bundled server file
CMD ["node", "dist/server.cjs"]
